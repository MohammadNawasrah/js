/**
 * A utility class for working with strings.
 * @class StringsUtils
 */
class stringUtils {
  /**
   * @param {string} string
   * @returns {string}
   */
  static removeNonWord(string) {
    return string.replace(/[^0-9a-zA-Z\xC0-\xFF \-]/g, "");
  }
}

files. */

  module.exports = {
    stringUtils,
  };
}
